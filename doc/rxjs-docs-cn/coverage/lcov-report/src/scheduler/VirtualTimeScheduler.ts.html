  public flush(): void {
 
    const {actions, maxFrames} = this;
    let error: any, action: AsyncAction<any>;
 
    while ((action = actions.shift()) && (this.frame = action.delay) <= maxFrames) {
      if (error = action.execute(action.state, action.delay)) {

        action.unsubscribe();
      }
      throw error;
    }
  }
